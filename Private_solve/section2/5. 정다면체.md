#### 내 풀이

```

n, m = map(int, input().split())
arr = []
arr2 = []
cnt = 0

for i in range(1,n+1):
    for j in range(1,m+1):
        arr.append(i+j)


for i in range(1, max(arr)):
    if(arr.count(i)>=cnt):
        if(arr.count(i)==cnt):
            arr2.append(i)
            cnt = arr.count(i)
            continue
        arr2.pop()
        arr2.append(i)
        cnt = arr.count(i)

arr2.sort()
for i in range(len(arr2)):
    print(arr2[i], end=' ')
    
```
